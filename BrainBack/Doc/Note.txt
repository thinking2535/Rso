// Map 사용 인근 2개 노드 연상
void CNeuronIn::Load(double In_)
{
	auto it = _Sensors.lower_bound(In_);
	if (it != _Sensors.end())
	{
		it->second->Load(In_);

		if (it->first == In_)
			return;

		if (it != _Sensors.begin())
		{
			--it;
			it->second->Load(In_);
		}
	}
	else
	{
		auto rb = _Sensors.rbegin();
		if (rb != _Sensors.rend())
			rb->second->Load(In_);
	}
}
